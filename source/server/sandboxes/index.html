Sandboxes

This module provides a method for creating sandboxed DOMs (using jsdom). It augments the DOM with a functional WebSocket using ws, and generally makes spinning up arbitrary DOM contexts a pleasant breeze.

config = require 'config'
Promise = require 'bluebird'
WebSocket = require 'socket.io/node_modules/socket.io-client/node_modules/engine.io-client/node_modules/ws/lib/WebSocket'

errors = require 'errors'
logging = require 'logging'

{jsdom} = require 'jsdom'
