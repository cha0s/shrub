If this is the first request made by a client, the cookie won't exist in req.headers.cookie. We normalize that inconsistency, so all consumers of the cookie will have a consistent interface on the first as well as subsequent requests.

(req, res, next) -> # If the client is in sync, awesome! return next() if req.signedCookies[key] is req.sessionID # Generate the cookie val = "s:" + signature.sign req.sessionID, cookie.cryptoKey cookieText = req.session.cookie.serialize key, val cookieObject = {} for kv in cookieText.split ';' [k, v] = kv.split '=' cookieObject[k.trim()] = v # Pull out junk that only makes sense en route to client. delete cookieObject['Path'] delete cookieObject['HttpOnly'] # Rebuild the cookie string. cookieText = '' for k, v of cookieObject cookieText += '; ' if cookieText cookieText += k + '=' + v # Commit the session before offering the cookie, otherwise it # wouldn't actually be pointing at anything yet. req.session.save (error) -> next error if error? req.signedCookies[key] = req.sessionID req.headers.cookie = cookieText next() ]
